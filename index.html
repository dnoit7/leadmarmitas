function importList() { const rawText = document.getElementById("rawList").value; // Divide a lista em linhas let lines = rawText.split("\n"); // Processa cada linha removendo números, hífens e espaços excessivos let cleanedNames = lines .map(line => line .replace(/^\s*\d+\s*[-.)]*\s*/i, "") // remove "1 -", "2)", "3.", etc. .trim() ) .filter(name => name.length > 0); // remove linhas vazias const container = document.getElementById("playersContainer"); container.innerHTML = ""; // Limpa campos antigos cleanedNames.forEach((name, index) => { const div = document.createElement("div"); div.className = "player"; div.innerHTML = ` `; container.appendChild(div); }); }
